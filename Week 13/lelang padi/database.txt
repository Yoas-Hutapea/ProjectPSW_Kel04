Create Database lelang_padi;
Create Table role(
id INT NOT NULL AUTO_INCREMENT,
rele_name VARCHAR(250),
PRIMARY KEY (id)
);
Create Table users (
id INT NOT NULL AUTO_INCREMENT,
nama Varchar(250),
no_hp Varchar(15),
no_rekening Varchar(20),
email Varchar(250),
alamat Varchar(250),
PASSWORD Varchar(250),
role_id int,
PRIMARY KEY(id),
FOREIGN KEY (role_id) REFREANCES role(id)
)
